resolve: {
	authentication: function (basicAuth, $timeout) {
		'use strict';
		if (!basicAuth.isAuthenticated()) {
			$timeout(function () {
				basicAuth.logout('/examples-spa/login/pageform');
				// $state.go('examples-spa_login_pageform'); //or use this
			}, 0);
		}
	}
}
